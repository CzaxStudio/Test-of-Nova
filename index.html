Security Header Audit

url = "https://example.com"
resp = http("GET", url)
hdrs = ["strict-transport-security","csp"]
each h : hdrs {
if httpHeader(resp, h) == "" { Trace("Missing: " + h) }
}

Mission Report

m = missionStart("Web Audit", "target.com")
